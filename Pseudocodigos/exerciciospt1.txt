Algoritmo "Desconto no preço do produto"

var
    preco, desconto, novoPreco: real

início
    leia preco
    
    desconto <- preco *  0.1
    novoPreco <- preco - desconto
    
    escreva novoPreco

fim


Algoritmo "Salário a ser recebido"

var
    salarioBase, gratificacao, salarioGratificacao, imposto, salarioRecebido: real

início
    leia salarioBase

    gratificacao <- salarioBase * 0.05
    salarioGratificacao <- salarioBase + gratificacao
    imposto <- salarioGratificacao * 0.07
    salarioRecebido <- salarioGratificacao - imposto

    escreva salarioRecebido

fim
