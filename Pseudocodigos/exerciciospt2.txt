Algoritmo " Calcula comissão"

var
    salarioFixo, valorVendas, comissao, salarioFinal : real

início
    leia salarioFixo, valorVendas

    comissao <- valorVendas * 0.04
    salarioFinal <- salarioFixo + comissao
    
    escreva comissao, salarioFinal

fim


Algoritmo "Calcula média ponderada"

var
    nota1, nota2, nota3, peso1, peso2, peso3, calculoNotas, somaPesos, mediaPonderada

início

    leia nota1, nota2, nota3, peso1, peso2, peso3

    calculoNotas <- nota1 * peso1 + nota2 * peso2 + nota3 * peso3
    somaPesos <- peso1 + peso2 + peso3
    mediaPonderada <- calculoNotas/somaPesos

    escreva mediaPonderada

fim


Algoritmo "Calcula área de triângulo"

var
    base, altura, area: real

início
    leia base, altura

    area <- (base * altura)/2

    escreva area

fim



Algoritmo "Calcula área do quadrado"

var
    lado, area: real

início
    leia lado

    area <- lado * lado

    escreva area

fim


Algoritmo "Converte medidas"

var 
    pes, polegadas, jardas, milhas: inteiro 

início
    leia pes

    polegadas <- pes * 12
    jardas <- pes / 3
    milhas <- jardas / 1760

    escreva polegadas, jardas, milhas

fim
