Algoritmo "Calcula idade"

var
    anoNascimento, anoAtual, idadeAnos, idadeMeses, idadeDias, idadeSemanas: inteiro

início
    leia anoNascimento, anoAtual

    idadeAnos <- anoAtual - anoNascimento
    idadeMeses <- idadeAnos * 12
    idadeDias <- idadeAnos * 365
    idadeSemanas <- idadeAnos * 52

    escreva idadeAnos, idadeMeses, idadeDias, idadeSemanas

fim


Algoritmo "Calcula custos de vendas de carros"

var
    precoFabrica, percentualLucro, percentualImpostos, lucro, impostos, precoFinal: real

início 
    leia precoFabrica, percentualLucro, percentualImpostos

    lucro <- precoFabrica * (percentualLucro / 100)
    impostos <- precoFabrica * (percentualImpostos / 100)
    precoFinal <- precoFabrica + lucro + impostos

    escreva precoFinal

fim


Algoritmo "Calcula salário"

var
    horasTrabalhadas, valorSalario, salarioBruto, imposto, salarioRecebido: real

início
    leia horasTrabalhadas, valorSalario
    
    valorHoraTrabalhada <- 0.1 * valorSalario
    salarioBruto <- horasTrabalhadas * valorHoraTrabalhada
    imposto <- salarioBruto * 0.03
    salarioRecebido <- salarioBruto - imposto
    
    escreva salarioRecebido

fim


Algoritmo "Calcula preço quilowatt"

var 
    valorSalario, qtdQuilowatt, valorPagar, valorPagarDesconto : real

início
    
    leia valorSalario, qtdQuilowatt
    
    valorQuilowatt <- valorSalario / 5
    valorPagar <- qtdQuilowatt * valorQuilowatt
    valorPagarDesconto <- valorPagar - (valorPagar * 0.15)
    
    escreva valorQuilowatt, valorPagar, valorPagarDesconto

fim


Algoritmo "Calcula Fahrenheit"

var
    celsius, fahrenheit: inteiro

início

    leia celsius

    fahrenheit <- ((9 * celsius)/5) + 32

    escreva fahrenheit

fim
